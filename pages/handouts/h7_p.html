Union Types



What is the output of the following program:

 // Union Types

 #include <iostream>
 #include <iomanip>
 using namespace std;

 enum hand {minute, hour};

 typedef struct {
     enum hand timer;
     union {
         int m;
         struct {
             int h;
             int m;
         } hourly;
     } time;
 } Counter;

 void tick(Counter* t) {

     switch(t->timer) {
     case minute: (t->time.m)++; break;
     case hour:   (t->time.hourly.m)++;
                  if (t->time.hourly.m == 60) {
                      (t->time.hourly.h)++;
                      t->time.hourly.m = 0;
                      if (t->time.hourly.h == 24)
                          t->time.hourly.h = 0;
                  }
                  break;
     }
 }

 void display(const Counter *t) {

     cout << setfill('0');
     switch(t->timer) {
     case minute:
         cout << setw(5) << t->time.m << endl;
         break;
     case hour:
         cout << setw(2) << t->time.hourly.h << ':' 
          << setw(2) << t->time.hourly.m << endl;
         break;
     }
     cout << setfill(' ');
 }

 int main() {
     int i;
     Counter stopwatch, clock;

     stopwatch.timer = minute;
     stopwatch.time.m = 0;
     clock.timer = hour;
     clock.time.hourly.h = 23;
     clock.time.hourly.m = 58;
     for (i = 0; i < 3; i++) {
         tick(&stopwatch);
         tick(&clock);
         display(&stopwatch);
         display(&clock);
     }
 }
