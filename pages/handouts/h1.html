Pre-Processor Macros



Consider the following source file

 // Handout on Macros

 #include <iostream>

 #define F0 1058
 #define F(n) ((n) + F0)
 #define NKEYS 5

 int main( ) {
     std::cout << "Key Value" << std::endl;
     for (int i = 0; i < NKEYS; i++)
         std::cout << "n = " << i + 1 << " F(n) = " << F(i + 1) << std::endl; 
 }
