Abstract Base Classes



Complete the following program so that the program shown below will generate the output shown on its right. 


 #include <iostream>
 #include <cstring>
 using namespace std;

 class IsDisplayable {
   public:
     virtual void display(ostream& os) const = 0;
 };

 ostream& operator<<(ostream& os, const IsDisplayable& d) {
     d.display(os);
     return os;
 }

 class Quote : public IsDisplayable {
     char* text;
   public:
     Quote(const char* t = "") {
         text = new char[strlen(t) + 1];
         strcpy(text, t);
     }
     Quote(const Quote& q) {
         text = NULL;
         *this = q;
     }
     Quote& operator=(const Quote& q) {
         if (this != &q) {
             if (text) delete [] text;
             text = new char[strlen(q.text) + 1];
             strcpy(text, q.text);
         }
         return *this;
     }
     ~Quote() {
         delete [] text;
     }









 };
