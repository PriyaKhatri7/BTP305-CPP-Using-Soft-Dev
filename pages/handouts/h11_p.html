Bit-Wise Expressions



What is the output of the following program:

 // Bit-Wise Operators - Practice
 // bits.cpp

 #include <iostream>
 using namespace std;

 unsigned x(unsigned a, unsigned b) {
     unsigned c = 0u;

     while (b) {
         if (b & 1)
             c += a;
         a <<= 1;
         b >>= 1;
     }
     return c;
 }

 int main () {

     cout << 27u << " \? " << 2u << " = " << x(27u, 2u) << endl;
     cout <<  7u << " \? " << 3u << " = " << x(7u,  3u) << endl;
 }
