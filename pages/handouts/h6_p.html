What is the output of the following program:

 // Multi-Dimensional Arrays - Handout

 #include <iostream>
 #include <iomanip>
 using namespace std;
 #define COLS 3
 #define ROWS 4

 void postMul(int y[], int m[][COLS], int x[], int r) {
     int i, j;

     for (i = 0; i < r; i++) {
         y[i] = 0;
         for (j = 0; j < COLS; j++)
             y[i] += m[i][j] * x[j];
     }
 }

 void showFullLine(int a, int b[], int c, int cols) {
     int i;

     cout << "| " << setw(2) << a << " |  |";
     for (i = 0; i < cols; i++)
         cout << " " << setw(2) << b[i] << " ";
     cout << "|| " << setw(2) << c << " |\n";
 }

 void showPartLine(int a, int b[], int cols) {
     int i;

     cout << "| " << setw(2) << a << " |  |";
     for (i = 0; i < cols; i++)
         cout << " " << setw(2) << b[i] << " ";
     cout << "|\n";
 }

 void showEndLine(int c, int cols) {
     int i;

     cout << "         ";
     for (i = 0; i < cols; i++)
         cout << "    ";
     cout << " | " << setw(2) << c << " |\n";
 }

 void display(int a[], int b[][COLS], int c[], int rows, int cols) {

     int i;

     for (i = 0; i < rows; i++)
         if (i < cols)
             showFullLine(a[i], b[i], c[i], cols);
         else
             showPartLine(a[i], b[i], cols);
     for (i = rows; i < cols; i++)
         showEndLine(c[i], cols);
 }

 int main () {

     int y[ROWS],
         m[ROWS][COLS] = { 1, 0, 0, 1, 2, 0, 1, 3, 2, 1, 2, 0},
         x[COLS] = {1, 2, 1},
         i;

     postMul(y, m, x, ROWS);
     display(y, m, x, ROWS, COLS);
 }
